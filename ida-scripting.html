ADDRESS & CURSOR NAVIGATION
idc.here() NAV
▶
def here() -> int
Returns the address currently under the cursor in IDA's disassembly view. Equivalent to idc.get_screen_ea(). The most commonly used function when writing interactive scripts.
Returns: int — linear address (ea)
>_ EXAMPLE
ea = idc.here() print(hex(ea)) # e.g. 0x140001000 # Jump cursor to address idc.jumpto(ea)
idc.jumpto() NAV
▶
def jumpto(ea: int) -> bool
Moves the cursor in IDA's view to the given address. Useful in interactive scripts to navigate the user to a discovered location.
PARAM TYPE DESC
ea int Target linear address
Returns: bool — True on success
>_ EXAMPLE
# Jump to entry point idc.jumpto(idc.get_inf_attr(idc.INF_START_EA))
idc.next_head() / idc.prev_head() NAV
▶
def next_head(ea: int, maxea: int = 0xFFFFFFFFFFFFFFFF) -> int def prev_head(ea: int, minea: int = 0) -> int
Move to the next or previous instruction/data head. Skips bytes that belong to the current item (e.g. multi-byte instructions). Returns idc.BADADDR if no next/prev head exists within bounds.
>_ EXAMPLE
ea = idc.here() for _ in range(10): print(hex(ea), idc.GetDisasm(ea)) ea = idc.next_head(ea)
FUNCTION ENUMERATION
idautils.Functions() NAV
▶
def Functions(start: int = None, end: int = None) -> Iterator[int]
Yields the start address of every function in the binary (or in the given address range). The most fundamental loop in any IDA script.
>_ EXAMPLE
import idautils for func_ea in idautils.Functions(): name = idc.get_func_name(func_ea) print(hex(func_ea), name)
idaapi.get_func() NAV
▶
def get_func(ea: int) -> ida_funcs.func_t | None
Returns the func_t object containing metadata for the function that owns the given address. Key attributes: .start_ea, .end_ea, .size(), .flags.
PARAM TYPE DESC
ea int Any address within the function
Returns: func_t — or None if not inside a function
>_ EXAMPLE
fn = idaapi.get_func(idc.here()) if fn: print(f"start={hex(fn.start_ea)} end={hex(fn.end_ea)} size={fn.size()}")
idautils.FuncItems() NAV
▶
def FuncItems(ea: int) -> Iterator[int]
Yields the address of every instruction that belongs to the function containing ea. Handles non-contiguous chunks automatically.
>_ EXAMPLE
# Count instructions in current function insns = list(idautils.FuncItems(idc.here())) print(f"{len(insns)} instructions")
INSTRUCTIONS & DISASSEMBLY
idc.GetDisasm() NAV
▶
def GetDisasm(ea: int) -> str
Returns the disassembly text for the instruction at ea as a string, including mnemonic and operands (e.g. "mov rax, [rbp+var_8]").
>_ EXAMPLE
# Find all call instructions in a function for ea in idautils.FuncItems(idc.here()): disasm = idc.GetDisasm(ea) if disasm.startswith("call"): print(hex(ea), disasm)
idc.print_insn_mnem() / idc.get_operand_value() NAV
▶
def print_insn_mnem(ea: int) -> str def get_operand_value(ea: int, n: int) -> int
print_insn_mnem returns just the mnemonic (e.g. "mov"). get_operand_value returns the numeric value of operand n (0-indexed). For calls/jumps this gives the target address.
PARAM TYPE DESC
ea int Instruction address
n int Operand index (0 = first, 1 = second)
>_ EXAMPLE
ea = idc.here() mnem = idc.print_insn_mnem(ea) if mnem == "call": target = idc.get_operand_value(ea, 0) print("calls ->", idc.get_func_name(target))
idaapi.decode_insn() NAV
▶
def decode_insn(insn: insn_t, ea: int) -> int
Fully decodes the instruction at ea into an insn_t object, giving access to operand types, registers, immediate values, and more detail than the string-based API.
>_ EXAMPLE
insn = idaapi.insn_t() if idaapi.decode_insn(insn, idc.here()): print("size:", insn.size) for op in insn.ops: if op.type == idaapi.o_reg: print("reg operand:", op.reg)
CROSS-REFERENCES — XrefsTo / XrefsFrom
idautils.XrefsTo() XREF
▶
def XrefsTo(ea: int, flags: int = 0) -> Iterator[xref_t]
Yields all cross-references pointing to the given address. Each yielded item is an xref_t object with fields: .frm (from address), .to (to address), .type (xref type), .iscode (1 if code ref). Use flags=1 to include data references.
PARAM TYPE DESC
ea int Address to find references to
flags int 0 = code xrefs only; 1 = include data xrefs
Returns: Iterator[xref_t] — each has .frm, .to, .type, .iscode
>_ EXAMPLE — who calls this function?
import idautils ea = idc.here() print(f"[*] XrefsTo {hex(ea)}:") for xref in idautils.XrefsTo(ea, 1): caller = idc.get_func_name(xref.frm) or hex(xref.frm) kind = "CODE" if xref.iscode else "DATA" print(f" [{kind}] {hex(xref.frm)} <- {caller}")
xref.type values: fl_JN=18 (jump near), fl_CN=17 (call near), fl_F=21 (flow), dr_R=1 (data read), dr_W=2 (data write)
idautils.XrefsFrom() XREF
▶
def XrefsFrom(ea: int, flags: int = 0) -> Iterator[xref_t]
Yields all cross-references originating from the given address — the inverse of XrefsTo. Useful to see what a single instruction calls, jumps to, or reads from.
>_ EXAMPLE
for xref in idautils.XrefsFrom(idc.here(), 1): print(f" -> {hex(xref.to)} type={xref.type}")
idautils.CodeRefsTo() / CodeRefsFrom() XREF
▶
def CodeRefsTo(ea: int, flow: bool) -> Iterator[int] def CodeRefsFrom(ea: int, flow: bool) -> Iterator[int]
Simpler wrappers that yield just the address (not an xref_t object). Set flow=False to exclude sequential flow references and get only real branches/calls.
PARAM TYPE DESC
ea int Target/source address
flow bool True = include sequential flow; False = branches/calls only
>_ EXAMPLE
# All callers of current function (no flow refs) ea = idc.here() for caller in idautils.CodeRefsTo(ea, False): print(hex(caller), idc.get_func_name(caller))
idautils.DataRefsTo() / DataRefsFrom() XREF
▶
def DataRefsTo(ea: int) -> Iterator[int] def DataRefsFrom(ea: int) -> Iterator[int]
Yields addresses of code or data that reads/writes the given location (DataRefsTo), or addresses that this instruction reads from (DataRefsFrom). Essential for tracking global variable and string usage.
>_ EXAMPLE — find all places that read a global
global_ea = 0x140005000 for ref in idautils.DataRefsTo(global_ea): print(f"read at {hex(ref)} in {idc.get_func_name(ref)}")
CALLERS & CALLEES
Get all callers of a function XREF
▶
Pattern combining XrefsTo + iscode filter to get every function that calls a target — the single most common malware RE task (e.g. "who calls VirtualAlloc?").
>_ PATTERN — callers of an import
import idautils, idc def get_callers(func_name): ea = idc.get_name_ea_simple(func_name) if ea == idc.BADADDR: print("not found"); return for xref in idautils.XrefsTo(ea, 0): if xref.iscode: fn = idc.get_func_name(xref.frm) or hex(xref.frm) print(f" {hex(xref.frm)} [{fn}]") get_callers("VirtualAlloc") get_callers("CreateThread")
NAMES & LABELS
idc.get_func_name() / idc.get_name() NAME
▶
def get_func_name(ea: int) -> str def get_name(ea: int, gtn_flags: int = 0) -> str
get_func_name returns the name IDA assigned to the function starting at ea. get_name is broader — it returns any named location (function, label, import, export). Returns empty string if unnamed.
>_ EXAMPLE
print(idc.get_func_name(idc.here())) # "sub_140001000" print(idc.get_name(0x140001000)) # same or user-set label
idc.set_name() NAME
▶
def set_name(ea: int, name: str, flags: int = 0) -> bool
Assigns a name (label) to an address. Use flags=idc.SN_NOCHECK to skip duplicate-name validation, or flags=idc.SN_FORCE to allow renaming even if IDA would refuse.
PARAM TYPE DESC
ea int Address to label
name str New name string
flags int SN_NOCHECK, SN_FORCE, SN_PUBLIC, etc.
>_ EXAMPLE — bulk rename obfuscated stubs
for i, fn_ea in enumerate(idautils.Functions()): name = idc.get_func_name(fn_ea) if name.startswith("sub_"): idc.set_name(fn_ea, f"unknown_{i}", idc.SN_NOCHECK)
idc.get_name_ea_simple() NAME
▶
def get_name_ea_simple(name: str) -> int
Reverse lookup — resolves a name string to its address. Returns idc.BADADDR if not found. Perfect for locating imports, exports, or any named symbol.
>_ EXAMPLE
ea = idc.get_name_ea_simple("WinExec") if ea != idc.BADADDR: print("WinExec at", hex(ea)) # Also: idaapi.get_name_ea(BADADDR, name)
idautils.Names() NAME
▶
def Names() -> Iterator[Tuple[int, str]]
Yields (address, name) tuples for every named location in the IDB — functions, labels, imports, exports. Useful for building a full symbol table.
>_ EXAMPLE
for ea, name in idautils.Names(): if "crypt" in name.lower(): print(hex(ea), name)
COMMENTS
idc.set_cmt() / idc.get_cmt() NAME
▶
def set_cmt(ea: int, comment: str, rptble: bool) -> bool def get_cmt(ea: int, rptble: bool) -> str
Set or read a comment at an address. rptble=True for a repeatable comment (shown at all xrefs too); rptble=False for a local comment at this address only.
>_ EXAMPLE
idc.set_cmt(idc.here(), "decrypts config blob with RC4", False) # Repeatable — shows wherever this func is referenced idc.set_cmt(idc.here(), "SUSPICIOUS: allocates RWX memory", True)
READ / WRITE MEMORY
idc.get_bytes() / idaapi.get_bytes() MEM
▶
def get_bytes(ea: int, size: int, use_dbg: bool = False) -> bytes # Single-byte shortcuts: def get_wide_byte(ea) -> int # 1 byte def get_wide_word(ea) -> int # 2 bytes def get_wide_dword(ea) -> int # 4 bytes def get_qword(ea) -> int # 8 bytes
Reads raw bytes from the IDB (database, not live process unless debugging). use_dbg=True reads from the debuggee's live memory during a debug session. Returns bytes object or raises on unmapped memory.
>_ EXAMPLE — extract encoded payload
ea = idc.here() size = 256 data = idc.get_bytes(ea, size) print(data.hex()) # XOR decrypt in-script key = 0x41 dec = bytes([b ^ key for b in data]) print(dec)
SEGMENTS
idautils.Segments() / idc.get_segm_name() SEG
▶
def Segments() -> Iterator[int] # yields seg start ea def get_segm_name(ea: int) -> str def get_segm_start(ea: int) -> int def get_segm_end(ea: int) -> int
Enumerate all segments (PE sections) in the binary. Each segment has a name (.text, .data, etc.), start/end addresses, and permissions.
>_ EXAMPLE
for seg_ea in idautils.Segments(): name = idc.get_segm_name(seg_ea) start = idc.get_segm_start(seg_ea) end = idc.get_segm_end(seg_ea) print(f"{name} {hex(start)}-{hex(end)} size={end-start}")
PATCHING
idc.patch_byte() / patch_word() / patch_dword() PATCH
▶
def patch_byte(ea: int, value: int) -> bool def patch_word(ea: int, value: int) -> bool def patch_dword(ea: int, value: int) -> bool def patch_qword(ea: int, value: int) -> bool
Modify bytes in the IDB. Changes are tracked as "patches" and can be applied to the original binary via Edit → Patch program → Apply patches to input file.
⚠ Patches modify the IDB, not the file on disk — use "Apply patches to input file" to persist.
>_ EXAMPLE — NOP out an anti-debug check
ea = idc.here() # NOP a 2-byte jz short instruction idc.patch_byte(ea, 0x90) idc.patch_byte(ea+1, 0x90) idaapi.auto_mark_range(ea, ea+2, idaapi.AU_CODE) # reanalyze
STRINGS
idautils.Strings() / idc.get_strlit_contents() STR
▶
def Strings(default_setup: bool = True) -> StringList def get_strlit_contents(ea, length, strtype) -> bytes | None
idautils.Strings() returns a string-list object iterable as StringItem objects with .ea, .length, .type. Use get_strlit_contents to read an arbitrary string at any address.
>_ EXAMPLE — dump all strings and their xrefs
sc = idautils.Strings() for s in sc: val = str(s) if any(k in val.lower() for k in ["http", "cmd", "powershell"]): print(f"[!] {hex(s.ea)}: {val}") for xref in idautils.XrefsTo(s.ea, 0): print(f" used at {hex(xref.frm)}")
IMPORTS & EXPORTS
idaapi.get_import_module_qty() / enum_import_names() IMPORT
▶
def get_import_module_qty() -> int def get_import_module_name(i: int) -> str def enum_import_names(mod_idx: int, callback: callable) -> bool
Walk the PE import table. The callback receives (ea, name, ordinal) and must return True to continue enumeration.
>_ EXAMPLE — print all imports
import idaapi def imp_cb(ea, name, ord): print(f" {hex(ea)} {name or f'ord_{ord}'}") return True n = idaapi.get_import_module_qty() for i in range(n): mod = idaapi.get_import_module_name(i) print(f"[{mod}]") idaapi.enum_import_names(i, imp_cb)
idautils.Entries() EXPORT
▶
def Entries() -> Iterator[Tuple[int, int, int, str]]
Yields (index, ordinal, ea, name) for every export in the binary. Combine with XrefsTo to find which exports are unused or suspicious.
>_ EXAMPLE
for idx, ordinal, ea, name in idautils.Entries(): print(f"export #{ordinal} {name} @ {hex(ea)}")
CONTROL FLOW & BASIC BLOCKS
idaapi.FlowChart() / BasicBlock CFG
▶
def FlowChart(f: func_t, flags: int = 0) -> Iterator[BasicBlock]
Builds and yields BasicBlock objects for every basic block in a function. Each block has .start_ea, .end_ea, .succs(), and .preds() for successor/predecessor navigation.
>_ EXAMPLE — count blocks and find large ones
fn = idaapi.get_func(idc.here()) chart = idaapi.FlowChart(fn) blocks = list(chart) print(f"{len(blocks)} basic blocks") for bb in blocks: size = bb.end_ea - bb.start_ea if size > 200: print(f"big block: {hex(bb.start_ea)} ({size} bytes)")
TYPE INFORMATION
idaapi.get_tinfo() / apply_tinfo() TYPE
▶
def get_tinfo(ea: int) -> Tuple[bool, tinfo_t] def apply_tinfo(ea: int, tif: tinfo_t, flags: int) -> bool def parse_decl(tif: tinfo_t, til, decl: str, flags: int) -> bool
Read and apply type information (function prototypes, struct types) to addresses. Useful for applying known calling conventions or fixing IDA's type inference on obfuscated code.
>_ EXAMPLE — apply a prototype to a function
tif = idaapi.tinfo_t() til = idaapi.get_idati() idaapi.parse_decl(tif, til, "LPVOID __stdcall VirtualAlloc(LPVOID, SIZE_T, DWORD, DWORD);", idaapi.PT_SIL) idaapi.apply_tinfo(idc.here(), tif, idaapi.TINFO_DEFINITE)
IDB & MISC
idc.get_inf_attr() — binary metadata IDB
▶
def get_inf_attr(attr: int) -> int
Queries binary-level metadata from the IDB. Common attributes: INF_START_EA (entry point), INF_PROCNAME (processor), INF_FILETYPE, INF_LFLAGS (bitness flags).
>_ EXAMPLE
ep = idc.get_inf_attr(idc.INF_START_EA) bits = 64 if idc.get_inf_attr(idc.INF_LFLAGS) & idc.LFLG_64BIT else 32 print(f"Entry: {hex(ep)} Arch: {bits}-bit")
idaapi.auto_wait() / refresh_idaview_anyway() IDB
▶
def auto_wait() -> None def refresh_idaview_anyway() -> None
auto_wait() blocks until IDA's background analysis queue is empty — call this after bulk patching or adding functions before querying results. refresh_idaview_anyway() forces the UI to redraw.
>_ EXAMPLE
# After making many changes, wait for re-analysis idaapi.auto_mark_range(start, end, idaapi.AU_CODE) idaapi.auto_wait() idaapi.refresh_idaview_anyway()
idaapi.ask_yn() / idc.msg() UI
▶
def ask_yn(default: int, msg: str) -> int # 1=yes, 0=no, -1=cancel def msg(s: str) -> None # print to output window def ask_str(defval: str, hist: int, prompt: str) -> str
Interactive UI helpers. msg() is the IDA equivalent of print() — it writes to the Output window. ask_yn pops a Yes/No dialog. ask_str prompts for a string input.
>_ EXAMPLE
idc.msg("[*] analysis started\n") ans = idaapi.ask_yn(1, "Rename all sub_ functions?") if ans == 1: idc.msg("renaming...\n")
